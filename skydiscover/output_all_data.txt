 459 | def _load_fit_table(data_file: Path) -> tuple[np.ndarray, np.ndarray]:
 460 |     """Load cleaned data.tsv and return X=[normalized time, m0, M0, c], y.
 461 | 
 462 |     Cleaned files store one trajectory per adjacent X/Y column pair. Each
 463 |     column header contains m0, M0, and L0 metadata; only m0 and M0 are exposed
 464 |     as static ODE features. Units are intentionally ignored per the cleaned-data
 465 |     note: the leading numeric value is used as provided.
 466 |     """
 467 |     with data_file.open(newline="") as f:
 468 |         reader = csv.reader(f, delimiter="\t")
 469 |         header = next(reader)
 470 |         rows = [row for row in reader if row]
 471 | 
 472 |     features: list[list[float]] = []
 473 |     targets: list[float] = []
 474 |     curve_ids: list[int] = []
 475 |     curve_id = 0
 476 |     for col_idx in range(len(header) - 1):
 477 |         x_header = header[col_idx]
 478 |         y_header = header[col_idx + 1]
 479 |         if not (_is_x_column(x_header) and _is_y_column(y_header)):
 480 |             continue
 481 |         m0 = _parse_metadata_number(x_header, "m0")
 482 |         seed_m0 = _parse_metadata_number(x_header, "M0")
 483 |         if m0 is None or seed_m0 is None:
 484 |             continue
 485 |         for row in rows:
 486 |             if col_idx + 1 >= len(row):
 487 |                 continue
 488 |             time_value = row[col_idx].strip()
 489 |             response_value = row[col_idx + 1].strip()
 490 |             if (
 491 |                 not time_value
 492 |                 or not response_value
 493 |                 or time_value.lower() == "nan"
 494 |                 or response_value.lower() == "nan"
 495 |             ):
 496 |                 continue
 497 |             features.append([float(time_value), m0, seed_m0])
 498 |             targets.append(float(response_value))
 499 |             curve_ids.append(curve_id)
 500 |         curve_id += 1
 501 | 
 502 |     if not features:
 503 |         raise ValueError(f"No regression samples loaded from {data_file}")
 504 | 
 505 |     X = np.asarray(features, dtype=float)
 506 |     y = np.asarray(targets, dtype=float)
 507 |     time_norm = _minmax_normalize(X[:, 0])
 508 |     monomer_raw = X[:, 1]
 509 |     seed_raw = X[:, 2]
 510 |     y_norm = _minmax_normalize(y)

 512 |     # x3 is the concentration state c used by ODE RHS templates. During scoring
 513 |     # it is supplied from odeint's dynamic state, not read from candidate code.
 514 |     X_ode = np.column_stack([time_norm, monomer_raw, seed_raw, y_norm])
 515 |     return _remember_curve_ids(X_ode, np.asarray(curve_ids, dtype=int)), y_norm

 637 | def _gaussian_smooth_1d(values: np.ndarray, sigma: float = GAUSSIAN_SMOOTH_SIGMA) -> np.ndarray:
 638 |     values = np.asarray(values, dtype=float).ravel()
 639 |     if values.size < 3 or sigma <= 0.0:
 640 |         return values
 641 |     radius = max(1, int(round(3.0 * sigma)))
 642 |     offsets = np.arange(-radius, radius + 1, dtype=float)
 643 |     kernel = np.exp(-0.5 * (offsets / sigma) ** 2)
 644 |     kernel /= np.sum(kernel)
 645 |     padded = np.pad(values, (radius, radius), mode="edge")
 646 |     return np.convolve(padded, kernel, mode="valid")

 665 | def _normalized_mse_with_reference_variance(

 667 |     y_pred: np.ndarray,
 668 |     reference_values: np.ndarray,
 669 | ) -> float:
 670 |     """MSE normalized by full dataset scale, not by a flat validation segment."""

 694 | def _ode_predictions(
 695 |     rhs_fn: Any,
 696 |     X: np.ndarray,
 697 |     y_known: np.ndarray,
 698 |     theta: np.ndarray,
 699 | ) -> np.ndarray:
 700 |     """Integrate dc/dt=f(t, m0, M0, c) per cleaned X/Y curve."""
 701 |     import warnings
 702 | 
 703 |     from scipy.integrate import odeint
 704 | 
 705 |     X = np.asarray(X, dtype=float)
 706 |     y_known = np.asarray(y_known, dtype=float).reshape(-1)
 707 |     theta = np.asarray(theta, dtype=float).reshape(-1)
 708 |     predictions = np.full_like(y_known, np.nan, dtype=float)
 709 | 
 710 |     for curve_idx in _curve_indices(X):
 711 |         ordered_idx = curve_idx[np.argsort(X[curve_idx, 0])]
 712 |         times = X[ordered_idx, 0]
 713 |         observed = y_known[ordered_idx]
 714 |         static_features = [float(value) for value in X[ordered_idx[0], 1:-1]]
 715 | 
 716 |         if len(ordered_idx) == 1 or float(times[-1] - times[0]) <= 0.0:
 717 |             predictions[ordered_idx] = observed[0]
 718 |             continue
 719 | 
 720 |         def dc_dt(c_state: np.ndarray, time_value: float) -> float:
 721 |             c_value = float(np.asarray(c_state, dtype=float).reshape(-1)[0])
 722 |             value = rhs_fn(float(time_value), *static_features, c_value, *theta)
 723 |             value = float(np.asarray(value, dtype=float).reshape(-1)[0])
 724 |             if not np.isfinite(value):
 725 |                 raise FloatingPointError("non-finite ODE derivative")
 726 |             return value
 727 | 
 728 |         with warnings.catch_warnings():
 729 |             warnings.simplefilter("ignore")
 730 |             integrated = odeint(
 731 |                 dc_dt,
 732 |                 float(observed[0]),
 733 |                 times,
 734 |                 mxstep=1000,
 735 |             )
 736 |         curve_pred = np.asarray(integrated, dtype=float).reshape(-1)
 737 |         if curve_pred.shape != observed.shape or not np.all(np.isfinite(curve_pred)):
 738 |             raise FloatingPointError("non-finite ODE trajectory")
 739 |         predictions[ordered_idx] = curve_pred
 740 | 
 741 |     return predictions
